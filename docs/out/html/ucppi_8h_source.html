37 /*
38  * A macro represented in a compact form; simple tokens are represented
39  * by one byte, containing their number. Tokens with a string value are
40  * followed by the value (string finished by a 0). Macro arguments are
41  * followed by the argument number (in one byte -- thus implying a hard
42  * limit of 254 arguments (number 255 is for __VA_ARGS__).
43  */
44 struct comp_token_fifo {
45  size_t length;
46  size_t rp;
47  unsigned char *t;
48 };

52 /*
53  * S_TOKEN(x) checks whether x is a token type with an embedded string
54  * ttMWS(x) checks whether x is macro whitespace (space, comment...)
55  * ttWHI(x) checks whether x is whitespace (MWS or newline)
56  */
57 #define S_TOKEN(x) STRING_TOKEN(x)
58 #define ttMWS(x) ((x) == NONE || (x) == COMMENT || (x) == OPT_NONE)
59 #define ttWHI(x) (ttMWS(x) || (x) == NEWLINE)
