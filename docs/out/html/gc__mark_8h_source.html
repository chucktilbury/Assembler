47 /* A client supplied mark procedure. Returns new mark stack pointer. */
48 /* Primary effect should be to push new entries on the mark stack. */
49 /* Mark stack pointer values are passed and returned explicitly. */
50 /* Global variables describing mark stack are not necessarily valid. */
51 /* (This usually saves a few cycles by keeping things in registers.) */
52 /* Assumed to scan about GC_PROC_BYTES on average. If it needs to do */
53 /* much more work than that, it should do it in smaller pieces by */
54 /* pushing itself back on the mark stack. */
55 /* Note that it should always do some work (defined as marking some */
56 /* objects) before pushing more than one entry on the mark stack. */
57 /* This is required to ensure termination in the event of mark stack */
58 /* overflows. */
59 /* This procedure is always called with at least one empty entry on the */
60 /* mark stack. */
61 /* Currently we require that mark procedures look for pointers in a */
62 /* subset of the places the conservative marker would. It must be safe */
63 /* to invoke the normal mark procedure instead. */
64 /* WARNING: Such a mark procedure may be invoked on an unused object */
65 /* residing on a free list. Such objects are cleared, except for a */
66 /* free list link field in the first word. Thus mark procedures may */
67 /* not count on the presence of a type descriptor, and must handle this */
68 /* case correctly somehow. Also, a mark procedure should be prepared */
69 /* to be executed concurrently from the marker threads (the later ones */
70 /* are created only if the client has called GC_start_mark_threads() */
71 /* or started a user thread previously). */
72 typedef struct GC_ms_entry * (*GC_mark_proc)(GC_word * /* addr */,
73  struct GC_ms_entry * /* mark_stack_ptr */,
74  struct GC_ms_entry * /* mark_stack_limit */,
75  GC_word /* env */);

86 /* Object descriptors on mark stack or in objects. Low order two */
87 /* bits are tags distinguishing among the following 4 possibilities */
88 /* for the high order 30 bits. */
89 #define GC_DS_TAG_BITS 2
90 #define GC_DS_TAGS ((1 << GC_DS_TAG_BITS) - 1)
91 #define GC_DS_LENGTH 0 /* The entire word is a length in bytes that */
92  /* must be a multiple of 4. */
93 #define GC_DS_BITMAP 1 /* 30 (62) bits are a bitmap describing pointer */
94  /* fields. The msb is 1 if the first word */
95  /* is a pointer. */
96  /* (This unconventional ordering sometimes */
97  /* makes the marker slightly faster.) */
98  /* Zeroes indicate definite nonpointers. Ones */
99  /* indicate possible pointers. */
100  /* Only usable if pointers are word aligned. */
101 #define GC_DS_PROC 2
102  /* The objects referenced by this object can be */
103  /* pushed on the mark stack by invoking */
104  /* PROC(descr). ENV(descr) is passed as the */
105  /* last argument. */
106 #define GC_MAKE_PROC(proc_index, env) \
107  (((((env) << GC_LOG_MAX_MARK_PROCS) \
108  | (proc_index)) << GC_DS_TAG_BITS) | GC_DS_PROC)
109 #define GC_DS_PER_OBJECT 3 /* The real descriptor is at the */
110  /* byte displacement from the beginning of the */
111  /* object given by descr & ~GC_DS_TAGS. */
112  /* If the descriptor is negative, the real */
113  /* descriptor is at (*<object_start>) - */
114  /* (descr&~GC_DS_TAGS) - GC_INDIR_PER_OBJ_BIAS */
115  /* The latter alternative can be used if each */
116  /* object contains a type descriptor in the */
117  /* first word. */
118  /* Note that in the multi-threaded environments */
119  /* per-object descriptors must be located in */
120  /* either the first two or last two words of */
121  /* the object, since only those are guaranteed */
122  /* to be cleared while the allocation lock is */
123  /* held. */
124 #define GC_INDIR_PER_OBJ_BIAS 0x10

133 /* Handle nested references in a custom mark procedure. */
134 /* Check if obj is a valid object. If so, ensure that it is marked. */
135 /* If it was not previously marked, push its contents onto the mark */
136 /* stack for future scanning. The object will then be scanned using */
137 /* its mark descriptor. */
138 /* Returns the new mark stack pointer. */

140 /* Since this marks first, it makes progress even if there are mark */

143 /* for back pointer-based heap debugging. */
