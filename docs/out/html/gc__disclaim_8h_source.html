35 /* Register "proc" to be called on each object of "kind" ready to be */
36 /* reclaimed. If "proc" returns non-zero, the collector will not */
37 /* reclaim the object on this GC cycle ("proc" should not try to */
38 /* resurrect the object otherwise). Objects reachable from "proc" */
39 /* (including the referred closure object) will be protected from */
40 /* collection if "mark_from_all" is non-zero, but at the expense that */
41 /* long chains of objects will take many cycles to reclaim. */
42 /* Calls to GC_free() will free its argument without inquiring "proc". */
43 /* No-op in the leak-finding mode. */
44 GC_API void GC_CALL GC_register_disclaim_proc(int /*kind*/,
45  GC_disclaim_proc /*proc*/,
46  int /*mark_from_all*/) GC_ATTR_NONNULL(2);

54 /* Allocate "size" bytes which is finalized by "fc". This uses a */
55 /* dedicated object kind with a disclaim procedure, and is more */
56 /* efficient than GC_register_finalizer and friends. */
57 /* GC_init_finalized_malloc must be called before using this. */
58 /* The collector will reclaim the object during this GC cycle (thus, */
59 /* "proc" should not try to resurrect the object). The other objects */
60 /* reachable from "proc" (including the closure object in case it is */
61 /* a heap-allocated one) will be protected from collection. */
62 /* Note that GC_size (applied to such allocated object) returns a value */
63 /* slightly bigger than the specified allocation size, and that GC_base */
64 /* result points to a word prior to the start of the allocated object. */
65 /* The disclaim procedure is not invoked in the leak-finding mode. */
66 GC_API GC_ATTR_MALLOC GC_ATTR_ALLOC_SIZE(1) void * GC_CALL
67  GC_finalized_malloc(size_t /*size*/,
68  const struct GC_finalizer_closure * /*fc*/) GC_ATTR_NONNULL(2);
