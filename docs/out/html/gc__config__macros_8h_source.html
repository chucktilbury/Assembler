346 /* If we're on a platform on which we can't save call stacks, but */
347 /* gcc is normally used, we go ahead and define GC_ADD_CALLER. */
348 /* We make this decision independent of whether gcc is actually being */
349 /* used, in order to keep the interface consistent, and allow mixing */
350 /* of compilers. */
351 /* This may also be desirable if it is possible but expensive to */
352 /* retrieve the call chain. */
353 #if (defined(__linux__) || defined(__DragonFly__) || defined(__FreeBSD__) \
354  || defined(__FreeBSD_kernel__) || defined(__HAIKU__) \
355  || defined(__NetBSD__) || defined(__OpenBSD__) \
356  || defined(HOST_ANDROID) || defined(__ANDROID__)) \
357  && !defined(GC_CAN_SAVE_CALL_STACKS)
358 # define GC_ADD_CALLER
359 # if GC_GNUC_PREREQ(2, 95)
360  /* gcc knows how to retrieve return address, but we don't know */
361  /* how to generate call stacks. */
362 # define GC_RETURN_ADDR (GC_word)__builtin_return_address(0)
363 # if GC_GNUC_PREREQ(4, 0) && (defined(__i386__) || defined(__amd64__) \
364  || defined(__x86_64__) /* and probably others... */) \
365  && !defined(GC_NO_RETURN_ADDR_PARENT)
366 # define GC_HAVE_RETURN_ADDR_PARENT
367 # define GC_RETURN_ADDR_PARENT \
368  (GC_word)__builtin_extract_return_addr(__builtin_return_address(1))
369  /* Note: a compiler might complain that calling */
370  /* __builtin_return_address with a nonzero argument is unsafe. */
371 # endif
372 # else
373  /* Just pass 0 for gcc compatibility. */
374 # define GC_RETURN_ADDR 0
375 # endif
376 #endif /* !GC_CAN_SAVE_CALL_STACKS */
