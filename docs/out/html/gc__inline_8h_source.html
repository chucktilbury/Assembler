97 /* The ultimately general inline allocation macro. Allocate an object */
98 /* of size granules, putting the resulting pointer in result. Tiny_fl */
99 /* is a "tiny" free list array, which will be used first, if the size */
100 /* is appropriate. If granules argument is too large, we allocate with */
101 /* default_expr instead. If we need to refill the free list, we use */
102 /* GC_generic_malloc_many with the indicated kind. */
103 /* Tiny_fl should be an array of GC_TINY_FREELISTS void * pointers. */
104 /* If num_direct is nonzero, and the individual free list pointers */
105 /* are initialized to (void *)1, then we allocate num_direct granules */
106 /* directly using generic_malloc before putting multiple objects into */
107 /* the tiny_fl entry. If num_direct is zero, then the free lists may */
108 /* also be initialized to (void *)0. */
109 /* Note that we use the zeroth free list to hold objects 1 granule in */
110 /* size that are used to satisfy size 0 allocation requests. */
111 /* We rely on much of this hopefully getting optimized away in the */
112 /* num_direct = 0 case. */
113 /* Particularly, if granules argument is constant, this should generate */
114 /* a small amount of code. */
115 # define GC_FAST_MALLOC_GRANS(result,granules,tiny_fl,num_direct, \
116  kind,default_expr,init) \
117  do { \
118  if (GC_EXPECT((granules) >= GC_TINY_FREELISTS, 0)) { \
119  result = (default_expr); \
120  } else { \
121  void **my_fl = (tiny_fl) + (granules); \
122  void *my_entry = *my_fl; \
123  void *next; \
124  \
125  for (;;) { \
126  if (GC_EXPECT((GC_word)my_entry \
127  > (num_direct) + GC_TINY_FREELISTS + 1, 1)) { \
128  next = *(void **)(my_entry); \
129  result = (void *)my_entry; \
130  GC_FAST_M_AO_STORE(my_fl, next); \
131  init; \
132  GC_PREFETCH_FOR_WRITE(next); \
133  if ((kind) != GC_I_PTRFREE) { \
134  GC_end_stubborn_change(my_fl); \
135  GC_reachable_here(next); \
136  } \
137  GC_ASSERT(GC_size(result) >= (granules)*GC_GRANULE_BYTES); \
138  GC_ASSERT((kind) == GC_I_PTRFREE \
139  || ((GC_word *)result)[1] == 0); \
140  break; \
141  } \
142  /* Entry contains counter or NULL */ \
143  if ((GC_signed_word)my_entry - (GC_signed_word)(num_direct) <= 0 \
144  /* (GC_word)my_entry <= (num_direct) */ \
145  && my_entry != 0 /* NULL */) { \
146  /* Small counter value, not NULL */ \
147  GC_FAST_M_AO_STORE(my_fl, (char *)my_entry \
148  + (granules) + 1); \
149  result = (default_expr); \
150  break; \
151  } else { \
152  /* Large counter or NULL */ \
153  GC_generic_malloc_many(((granules) == 0? GC_GRANULE_BYTES : \
154  GC_RAW_BYTES_FROM_INDEX(granules)), \
155  kind, my_fl); \
156  my_entry = *my_fl; \
157  if (my_entry == 0) { \
158  result = (*GC_get_oom_fn())((granules)*GC_GRANULE_BYTES); \
159  break; \
160  } \
161  } \
162  } \
163  } \
164  } while (0)

169 /* Allocate n words (NOT BYTES). X is made to point to the result. */
170 /* This should really only be used if GC_all_interior_pointers is */
171 /* not set, or DONT_ADD_BYTE_AT_END is set. See above. */
172 /* Does not acquire lock. The caller is responsible for supplying */
173 /* a cleared tiny_fl free list array. For single-threaded */
174 /* applications, this may be a global array. */
175 # define GC_MALLOC_WORDS_KIND(result,n,tiny_fl,kind,init) \
176  do { \
177  size_t granules = GC_WORDS_TO_WHOLE_GRANULES(n); \
178  GC_FAST_MALLOC_GRANS(result, granules, tiny_fl, 0, kind, \
179  GC_malloc_kind(granules*GC_GRANULE_BYTES, kind), \
