37 /*
38  * Hash a string into an `unsigned' value. This function is derived
39  * from the hash function used in the ELF binary object file format
40  * hash tables. The result size is a 32-bit number if the `unsigned'
41  * type is big enough to hold 32-bit arbitrary numbers, a 16-bit number
42  * otherwise.
43  */
44 static unsigned hash_string(char *name)
45 {
46  unsigned h = 0;
47 
48  for (h = 0; *name; name ++) {
49  unsigned g;
50 
51  h = (h << 4) + *(unsigned char *)name;
52 #if UINT_MAX >= 0xffffffffU
53  g = h & 0xF0000000U;
54  h ^= (g >> 24);
55 #else
56  g = h & 0xF000U;
57  h ^= (g >> 12);
58 #endif
59  h &= ~g;
60  }
61  return h;
62 }

64 /*
65  * Each item in the table is a structure beginning with a `hash_item_header'
66  * structure. Those headers define binary trees such that all left-descendants
67  * (respectively right-descendants) of a given tree node have an associated
68  * hash value strictly smaller (respectively greater) than the hash value
69  * associated with this node.
70  *
71  * The `ident' field points to an array of char. The `sizeof(unsigned)'
72  * first `char' contain a copy of an `unsigned' value which is the hashed
73  * string, except the least significant bit. When this bit is set to 0,
74  * the node contains the unique item using that hash value. If the bit
75  * is set to 1, then there are several items with that hash value.
76  *
77  * When several items share the same hash value, they are linked together
78  * in a linked list by their `left' field. The node contains no data;
79  * it is a "fake item".
80  *
81  * The `char' following the hash value encode the item name for true items.
82  * For fake items, they contain the pointer to the first true item of the
83  * corresponding link list (suitably aligned).
84  *
85  * There are HTT_NUM_TREES trees; the items are sorted among trees by the
86  * lest significant bits of their hash value.
87  */

116  ((sizeof(unsigned) + sizeof(hash_item_header *) - 1) / \
117  sizeof(hash_item_header *)))

122 /*
123  * Find a node for the given hash value. If `father' is not NULL, fill
124  * `*father' with a pointer to the node's father.
125  * If the return value is NULL, then no existing node was found; if `*father'
126  * is also NULL, the tree is empty. If the return value is not NULL but
127  * `*father' is NULL, then the found node is the tree root.
128  *
129  * If `father' is not NULL, then `*leftson' is filled with 1 if the node
130  * was looked for as the father left son, 0 otherwise.
131  */
132 static hash_item_header *find_node(HTT *htt, unsigned u,
133  hash_item_header **father, int *leftson, int reduced)
134 {
135  hash_item_header *node = TREE(u);
136  hash_item_header *nodef = NULL;
137  int ls = 0;
138 
139  u &= ~1U;
140  while (node != NULL) {
141  unsigned v = *(unsigned *)(node->ident);
142  unsigned w = v & ~1U;
143 
144  if (u == w) break;
145  nodef = node;
146  if (u < w) {
147  node = node->left;
148  ls = 1;
149  } else {
150  node = node->right;
151  ls = 0;
152  }
153  }
154  if (father != NULL) {
155  *father = nodef;
156  *leftson = ls;
157  }
158  return node;
159 }

161 static void *internal_get(HTT *htt, char *name, int reduced)
162 {
163  unsigned u = hash_string(name), v;
164  hash_item_header *node = find_node(htt, u, NULL, NULL, reduced);
165 
166  if (node == NULL) return NULL;
167  v = *(unsigned *)(node->ident);
168  if ((v & 1U) == 0) {
169  return (strcmp(HASH_ITEM_NAME(node), name) == 0) ? node : NULL;
170  }
171  node = *(hash_item_header **)(node->ident + PTR_SHIFT);
172  while (node != NULL) {
173  if (strcmp(HASH_ITEM_NAME(node), name) == 0) return node;
174  node = node->left;
175  }
176  return NULL;
177 }

192  * Make an item identifier from its name and its hash value.
193  */
194 static char *make_ident(char *name, unsigned u)
195 {
196  size_t n = strlen(name) + 1;
197  char *ident = getmem(n + sizeof(unsigned));
198 
199  *(unsigned *)ident = u & ~1U;
200  memcpy(ident + sizeof(unsigned), name, n);
201  return ident;
202 }

204 /*
205  * Make an identifier for a fake item, pointing to a true item.
206  */
207 static char *make_fake_ident(unsigned u, hash_item_header *next)
208 {
209  char *ident = getmem(PTR_SHIFT + sizeof(hash_item_header *));
210 
211  *(unsigned *)ident = u | 1U;
212  *(hash_item_header **)(ident + PTR_SHIFT) = next;

216 /*
217  * Adding an item is straightforward:
218  * 1. look for its emplacement
219  * 2. if no node is found, use the item as a new node and link it to the tree
220  * 3. if a node is found:
221  * 3.1. if the node is real, check for name inequality, then create a
222  * fake node and assemble the two-element linked list
223  * 3.2. if the node is fake, look for the name in the list; if not found,
224  * add the node at the list end
225  */
226 static void *internal_put(HTT *htt, void *item, char *name, int reduced)
227 {
228  unsigned u = hash_string(name), v;
229  int ls;
230  hash_item_header *father;
231  hash_item_header *node = find_node(htt, u, &father, &ls, reduced);
232  hash_item_header *itemg = item;
233  hash_item_header *pnode = NULL;
234 
235  if (node == NULL) {
236  itemg->left = itemg->right = NULL;
237  itemg->ident = make_ident(name, u);
238  if (father == NULL) {
239  TREE(u) = itemg;
240  } else if (ls) {
241  father->left = itemg;
242  } else {
243  father->right = itemg;
244  }
245  return NULL;
246  }
247  v = *(unsigned *)(node->ident);
248  if ((v & 1U) == 0) {
249  if (strcmp(HASH_ITEM_NAME(node), name) == 0)
250  return node;
251  pnode = getmem(sizeof *pnode);
252  pnode->left = node->left;
253  pnode->right = node->right;
254  pnode->ident = make_fake_ident(u, node);
255  node->left = itemg;
256  node->right = NULL;
257  itemg->left = itemg->right = NULL;
258  itemg->ident = make_ident(name, u);
259  if (father == NULL) {
260  TREE(u) = pnode;
261  } else if (ls) {
262  father->left = pnode;
263  } else {
264  father->right = pnode;
265  }
266  return NULL;
267  }
268  node = *(hash_item_header **)(node->ident + PTR_SHIFT);
269  while (node != NULL) {
270  if (strcmp(HASH_ITEM_NAME(node), name) == 0) return node;
271  pnode = node;
272  node = node->left;
273  }
274  itemg->left = itemg->right = NULL;
275  itemg->ident = make_ident(name, u);
276  pnode->left = itemg;
277  return NULL;
278 }

292 /*
293  * A fake node subnode list has shrunk to one item only; make the
294  * node real again.
295  * fnode the fake node
296  * node the last remaining node
297  * father the fake node father (NULL if the fake node is root)
298  * leftson 1 if the fake node is a left son, 0 otehrwise
299  * u the hash value for this node
300  */
301 static void shrink_node(HTT *htt, hash_item_header *fnode,
302  hash_item_header *node, hash_item_header *father, int leftson,
303  unsigned u, int reduced)
304 {
305  node->left = fnode->left;
306  node->right = fnode->right;
307  if (father == NULL) {
308  TREE(u) = node;
309  } else if (leftson) {
310  father->left = node;
311  } else {
312  father->right = node;
313  }
314  freemem(fnode->ident);
315  freemem(fnode);
316 }

318 /*
319  * Deletion algorithm:
320  * 1. look for the node; if not found, exit
321  * 2. if the node is real:
322  * 2.1. check for equality; exit otherwise
323  * 2.2. delete the node
324  * 2.3. promote the leftest of right descendants or rightest of left
325  * descendants
326  * 3. if the node is fake:
327  * 3.1. check the list items for equality; exit otherwise
328  * 3.2. delete the correct item
329  * 3.3. if there remains only one item, supress the fake node
330  */
331 static int internal_del(HTT *htt, char *name, int reduced)
332 {
333  unsigned u = hash_string(name), v;
334  int ls;
335  hash_item_header *father;
336  hash_item_header *node = find_node(htt, u, &father, &ls, reduced);
337  hash_item_header *pnode, *fnode, *znode;
338  char *tmp;
339 
340  if (node == NULL) return 0;
341  v = *(unsigned *)(node->ident);
342  if ((v & 1U) != 0) {
343  fnode = node;
344  node = znode = *(hash_item_header **)(node->ident + PTR_SHIFT);
345  pnode = NULL;
346  while (node != NULL) {
347  if (strcmp(HASH_ITEM_NAME(node), name) == 0) break;
348  pnode = node;
349  node = node->left;
350  }
351  if (node == NULL) return 0;
352  if (pnode == NULL) {
353  /*
354  * We supress the first item in the list.
355  */
356  *(hash_item_header **)(fnode->ident + PTR_SHIFT) =
357  node->left;
358  if (node->left->left == NULL) {
359  shrink_node(htt, fnode, node->left,
360  father, ls, u, reduced);
361  }
362  } else {
363  pnode->left = node->left;
364  if (pnode->left == NULL && znode == pnode) {
365  shrink_node(htt, fnode, pnode,
366  father, ls, u, reduced);
367  }
368  }
369  } else {
370  if (strcmp(HASH_ITEM_NAME(node), name) != 0) return 0;
371  if (node->left != NULL) {
372  for (znode = node, pnode = node->left; pnode->right;
373  znode = pnode, pnode = pnode->right);
374  if (znode != node) {
375  znode->right = pnode->left;
376  pnode->left = node->left;
377  }
378  pnode->right = node->right;
379  } else if (node->right != NULL) {
380  for (znode = node, pnode = node->right; pnode->left;
381  znode = pnode, pnode = pnode->left);
382  if (znode != node) {
383  znode->left = pnode->right;
384  pnode->right = node->right;
385  }
386  pnode->left = node->left;
387  } else pnode = NULL;
388  if (father == NULL) {
389  TREE(u) = pnode;
390  } else if (ls) {
391  father->left = pnode;
392  } else {
393  father->right = pnode;
394  }
395  }
396  tmp = node->ident;
397  htt->deldata(node);
398  freemem(tmp);
399  return 1;
400 }

414 /*
415  * Apply `action()' on all nodes of the tree whose root is given as
416  * parameter `node'. If `wipe' is non-zero, the nodes are removed
417  * from memory.
418  */
419 static void scan_node(hash_item_header *node, void (*action)(void *), int wipe)
420 {
421  unsigned v;
422 
423  if (node == NULL) return;
424  scan_node(node->left, action, wipe);
425  scan_node(node->right, action, wipe);
426  v = *(unsigned *)(node->ident);
427  if ((v & 1U) != 0) {
428  hash_item_header *pnode, *nnode;
429 
430  for (pnode = *(hash_item_header **)(node->ident + PTR_SHIFT);
431  pnode != NULL; pnode = nnode) {
432  char *tmp = pnode->ident;
433 
434  nnode = pnode->left;
435  action(pnode);
436  if (wipe) freemem(tmp);
437  }
438  if (wipe) {
439  freemem(node->ident);
440  freemem(node);
441  }
442  } else {
443  char *tmp = node->ident;
444 
445  action(node);
446  if (wipe) freemem(tmp);
447  }
448 }
