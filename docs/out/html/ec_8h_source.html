25 /* Extensible cords are strings that may be destructively appended to. */
26 /* They allow fast construction of cords from characters that are */
27 /* being read from a stream. */
28 /*
29  * A client might look like:
30  *
31  * {
32  * CORD_ec x;
33  * CORD result;
34  * char c;
35  * FILE *f;
36  *
37  * ...
38  * CORD_ec_init(x);
39  * while(...) {
40  * c = getc(f);
41  * ...
42  * CORD_ec_append(x, c);
43  * }
44  * result = CORD_balance(CORD_ec_to_cord(x));
45  *
46  * If a C string is desired as the final result, the call to CORD_balance
47  * may be replaced by a call to CORD_to_char_star.
48  */

73 /* Append a character to an extensible cord. */

75  ((void)((x)[0].ec_bufptr == (x)[0].ec_buf + CORD_BUFSZ \
