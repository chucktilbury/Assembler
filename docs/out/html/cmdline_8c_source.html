153 void parseCmdLine(int argc, char** argv)
154 {
155  int index = 0;
156  const char* arg;
157 
158  // save the program name
159  cmds->fname = dup_string(argv[index++]);
160 
161  while(index < argc) {
162  // find the parameter
163  _cmd_line_elem* param = find_param(argv[index]);
164  if(param == NULL) {
165  // if it's not found, add it to the file list
166  add_file(argv[index++]);
167  continue;
168  }
169  else {
170  // check to see if a space separates the parameter from the arg
171  size_t len = strlen(param->param);
172  if(param->type == CL_TOGGLE || param->type == CL_CB_WO) {
173  // there is no arg, just index to the next one.
174  arg = NULL;
175  index++;
176  }
177  else if(strlen(argv[index]) > len) {
178  // there is no space
179  arg = &argv[index][len];
180  index++;
181  }
182  else {
183  // there is a space.
184  index++; // index the arg
185  arg = argv[index];
186  index++; // index the next param
187  }
188  }
189 
190  // parse the parameter arguments
191  switch(param->type) {
192  case CL_TOGGLE:
193  param->data.bval = param->data.bval? false: true;
194  param->flags |= CL_PRESENT;
195  break;
196  case CL_CB_WO:
197  param->data.without();
198  param->flags |= CL_PRESENT;
199  break;
200  case CL_NUMBER:
201  param->data.num = strtol(arg, NULL, 10);
202  param->flags |= CL_PRESENT;
203  break;
204  case CL_STRING:
205  param->data.str = dup_string(arg);
206  param->flags |= CL_PRESENT;
207  break;
208  case CL_CB_W:
209  param->data.with(arg);
210  param->flags |= CL_PRESENT;
211  break;
212  default:
213  fprintf(stderr, "internal cmd error: unknown type: %d\n", param->type);
214  exit(1);
215  }
216  }
217 
218  // verify that required params are present
219  for(_cmd_line_elem* cel = cmds->cfirst; cel != NULL; cel = cel->next) {
220  if((cel->flags & CL_REQD) && !(cel->flags & CL_PRESENT)) {
221  fprintf(stderr, "cmd line error: required parameter '%s' is not found\n", cel->param);
222  showUseCmdLine();
223  }
224  }
225 
226  if(cmds->flags & CL_NO_FL) {
227  if(cmds->ffirst != NULL) {
228  fprintf(stderr, "cmd line error: extra parameters on command line\n");
229  showUseCmdLine();
230  }
231  }
232  else {
233  if(cmds->flags & (CL_FL_REQD|CL_FL_ONE)) {
234  if(cmds->ffirst == NULL) {
235  fprintf(stderr, "cmd line error: required file list is not found\n");
236  showUseCmdLine();
237  }
238  }
239 
240  if(cmds->flags & (CL_FL_ONE)) {
241  if(cmds->ffirst != cmds->flast) {
242  fprintf(stderr, "cmd line error: exactly one file name required\n");
243  showUseCmdLine();
244  }
245  }
246  }
247 }
