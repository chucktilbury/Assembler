25 /*
26  * This implements standard-conforming allocators that interact with
27  * the garbage collector. Gc_allocator<T> allocates garbage-collectible
28  * objects of type T. Traceable_allocator<T> allocates objects that
29  * are not themselves garbage collected, but are scanned by the
30  * collector for pointers to collectible objects. Traceable_alloc
31  * should be used for explicitly managed STL containers that may
32  * point to collectible objects.
33  *
34  * This code was derived from an earlier version of the GNU C++ standard
35  * library, which itself was derived from the SGI STL implementation.
36  *
37  * Ignore-off-page allocator: George T. Talbot
38  */

88 // In the following GC_Tp is GC_true_type if we are allocating a
89 // pointer-free object.
90 template <class GC_Tp>
91 inline void * GC_selective_alloc(size_t n, GC_Tp, bool ignore_off_page) {
92  void *obj = ignore_off_page ? GC_MALLOC_IGNORE_OFF_PAGE(n) : GC_MALLOC(n);
93  if (0 == obj)
94  GC_ALLOCATOR_THROW_OR_ABORT();
95  return obj;
96 }

111 // Now the public gc_allocator<T> class.
112 template <class GC_Tp>
113 class gc_allocator {
114 public:
115  typedef size_t size_type;
116  typedef ptrdiff_t difference_type;
117  typedef GC_Tp* pointer;
118  typedef const GC_Tp* const_pointer;
119  typedef GC_Tp& reference;
120  typedef const GC_Tp& const_reference;
121  typedef GC_Tp value_type;
122 
123  template <class GC_Tp1> struct rebind {
124  typedef gc_allocator<GC_Tp1> other;
125  };
126 
127  gc_allocator() GC_NOEXCEPT {}
128  gc_allocator(const gc_allocator&) GC_NOEXCEPT {}
129 # ifndef GC_NO_MEMBER_TEMPLATES
130  template <class GC_Tp1> GC_ATTR_EXPLICIT
131  gc_allocator(const gc_allocator<GC_Tp1>&) GC_NOEXCEPT {}
132 # endif
133  ~gc_allocator() GC_NOEXCEPT {}
134 
135  pointer address(reference GC_x) const { return &GC_x; }
136  const_pointer address(const_reference GC_x) const { return &GC_x; }
137 
138  // GC_n is permitted to be 0. The C++ standard says nothing about what
139  // the return value is when GC_n == 0.
140  GC_Tp* allocate(size_type GC_n, const void* = 0) {
141  GC_type_traits<GC_Tp> traits;
142  return static_cast<GC_Tp *>(GC_selective_alloc(GC_n * sizeof(GC_Tp),
143  traits.GC_is_ptr_free,
144  false));
145  }
146 
147  void deallocate(pointer __p, size_type /* GC_n */) GC_NOEXCEPT
148  { GC_FREE(__p); }
149 
150  size_type max_size() const GC_NOEXCEPT
151  { return static_cast<size_t>(-1) / sizeof(GC_Tp); }
152 
153  void construct(pointer __p, const GC_Tp& __val) { new(__p) GC_Tp(__val); }
154  void destroy(pointer __p) { __p->~GC_Tp(); }
155 };
