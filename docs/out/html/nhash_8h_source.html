33 /*
34  * Each item stored in the hash table should be a structure beginning
35  * with the following header.
36  */
37 typedef struct hash_item_header_ {
38  char *ident;
39  struct hash_item_header_ *left, *right;
40 } hash_item_header;
41 
42 /*
43  * This macro takes as argument a pointer to a hash table item (a
44  * structure beginning with `hash_item_header') and returns a pointer to
45  * the item name. This name should be considered as read-only. The
46  * retrieved pointer can become invalid whenever a new item is inserted
47  * in or removed from the table.
48  */
49 #define HASH_ITEM_NAME(s) (((hash_item_header *)(s))->ident + sizeof(unsigned))

51 /*
52  * Number of lists for the primary hash step. Can be reduced to save more
53  * memory, or increased to speed things up. It should be a power of 2
54  * greater or equal than 2 and smaller than UINT_MAX.
55  */
56 #define HTT_NUM_TREES 128
57 
58 /*
59  * Type for a hash table.
60  */
61 typedef struct {
62  void (*deldata)(void *);
63  hash_item_header *tree[HTT_NUM_TREES];
64 } HTT;
65 
66 /*
67  * Type for a reduced version of HTT with only two binary trees. That
68  * version has a lower initialization time and is suitable for situation
69  * where only a limited number of elements will be stored, but new tables
70  * need frequent initializations.
71  */
72 typedef struct {
73  void (*deldata)(void *);
74  hash_item_header *tree[2];
75 } HTT2;

77 /*
78  * Initialize a hash table. The `deldata' parameter should point to a
79  * function which will be invoked on any item removed from the table;
80  * that function should take care of the release of memory allocated for
81  * that item (except the hash_item_header contents, which are handled
82  * internally).
83  */
84 void HTT_init(HTT *htt, void (*deldata)(void *));
85 
86 /*
87  * Link an item into the hash table under the given name. If another
88  * item of identical name is already present in the table, a pointer to
89  * that item is returned; otherwise, the new item is linked into the
90  * table and NULL is returned. The object pointed to by `item' is
91  * linked from the table, but not the string pointed to by `name'.
92  */
93 void *HTT_put(HTT *htt, void *item, char *name);

95 /*
96  * Retrieve an item by name from the hash table. NULL is returned if

98  */

101 /*
102  * Remove an item from the hash table. 1 is returned if the item was
103  * removed, 0 if it was not found.
104  */
105 int HTT_del(HTT *htt, char *name);

107 /*
108  * For all items stored within the hash table, invoke the provided
