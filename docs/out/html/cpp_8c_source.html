589 #ifdef UCPP_MMAP
590 /*
591  * We open() the file, then fdopen() it and fseek() to its end. If the
592  * fseek() worked, we try to mmap() the file, up to the point where we
593  * arrived.
594  * On an architecture where end-of-lines are multibytes and translated
595  * into single '\n', bad things could happen. We strongly hope that, if
596  * we could fseek() to the end but could not mmap(), then we can get back.
597  */
598 static void *find_file_map;
599 static size_t map_length;
600 
601 FILE *fopen_mmap_file(char *name)
602 {
603  FILE *f;
604  int fd;
605  long l;
606 
607  find_file_map = 0;
608  fd = open(name, O_RDONLY, 0);
609  if (fd < 0) return 0;
610  l = lseek(fd, 0, SEEK_END);
611  f = fdopen(fd, "r");
612  if (!f) {
613  close(fd);
614  return 0;
615  }
616  if (l < 0) return f; /* not seekable */
617  map_length = l;
618  if ((find_file_map = mmap(0, map_length, PROT_READ,
619  MAP_PRIVATE, fd, 0)) == MAP_FAILED) {
620  /* we could not mmap() the file; get back */
621  find_file_map = 0;
622  if (fseek(f, 0, SEEK_SET)) {
623  /* bwaah... can't get back. This file is cursed. */
624  fclose(f);
625  return 0;
626  }
627  }
628  return f;
629 }

980 /*
981  * The #if directive. This function parse the expression, performs macro
982  * expansion (and handles the "defined" operator), and call eval_expr.
983  * return value: 1 if the expression is true, 0 if it is false, -1 on error.
984  */
985 static int handle_if(struct lexer_state *ls)
986 {
987  struct token_fifo tf, tf1, tf2, tf3, *save_tf;
988  long l = ls->line;
989  unsigned long z;
990  int ret = 0, ltww = 1;
991 
992  /* first, get the whole line */
993  tf.art = tf.nt = 0;
994  tf.t = NULL;
995  while (!next_token(ls) && ls->ctok->type != NEWLINE) {
996  struct token t;
997 
998  if (ltww && ttMWS(ls->ctok->type)) continue;
999  ltww = ttMWS(ls->ctok->type);
1000  t.type = ls->ctok->type;
1001  t.line = l;
1002  if (S_TOKEN(ls->ctok->type)) {
1003  t.name = sdup(ls->ctok->name);
1004  throw_away(ls->gf, t.name);
1005  }
1006  aol(tf.t, tf.nt, t, TOKEN_LIST_MEMG);
1007  }
1008  if (ltww && tf.nt) if ((-- tf.nt) == 0) freemem(tf.t);
1009  if (tf.nt == 0) {
1010  error(l, "void condition for a #if/#elif");
1011  return -1;
1012  }
1013  /* handle the "defined" operator */
1014  tf1.art = tf1.nt = 0;
1015  while (tf.art < tf.nt) {
1016  struct token *ct, rt;
1017  struct macro *m;
1018  size_t nidx, eidx;
1019 
1020  ct = tf.t + (tf.art ++);
1021  if (ct->type == NAME && !strcmp(ct->name, "defined")) {
1022  if (tf.art >= tf.nt) goto store_token;
1023  nidx = tf.art;
1024  if (ttMWS(tf.t[nidx].type))
1025  if (++ nidx >= tf.nt) goto store_token;
1026  if (tf.t[nidx].type == NAME) {
1027  eidx = nidx;
1028  goto check_macro;
1029  }
1030  if (tf.t[nidx].type != LPAR) goto store_token;
1031  if (++ nidx >= tf.nt) goto store_token;
1032  if (ttMWS(tf.t[nidx].type))
1033  if (++ nidx >= tf.nt) goto store_token;
1034  if (tf.t[nidx].type != NAME) goto store_token;
1035  eidx = nidx + 1;
1036  if (eidx >= tf.nt) goto store_token;
1037  if (ttMWS(tf.t[eidx].type))
1038  if (++ eidx >= tf.nt) goto store_token;
1039  if (tf.t[eidx].type != RPAR) goto store_token;
1040  goto check_macro;
1041  }
1042  store_token:
1043  aol(tf1.t, tf1.nt, *ct, TOKEN_LIST_MEMG);
1044  continue;
1045 
1046  check_macro:
1047  m = get_macro(tf.t[nidx].name);
1048  rt.type = NUMBER;
1049  rt.name = m ? "1L" : "0L";
1050  aol(tf1.t, tf1.nt, rt, TOKEN_LIST_MEMG);
1051  tf.art = eidx + 1;
1052  }
1053  freemem(tf.t);
1054  if (tf1.nt == 0) {
1055  error(l, "void condition (after expansion) for a #if/#elif");
1056  return -1;
1057  }
1058 
1059  /* perform all macro substitutions */
1060  tf2.art = tf2.nt = 0;
1061  save_tf = ls->output_fifo;
1062  ls->output_fifo = &tf2;
1063  while (tf1.art < tf1.nt) {
1064  struct token *ct;
1065 
1066  ct = tf1.t + (tf1.art ++);
1067  if (ct->type == NAME) {
1068  struct macro *m = get_macro(ct->name);
1069 
1070  if (m) {
1071  if (substitute_macro(ls, m, &tf1, 0,
1072 #ifdef NO_PRAGMA_IN_DIRECTIVE
1073  1,
1074 #else
1075  0,
1076 #endif
1077  ct->line)) {
1078  ls->output_fifo = save_tf;
1079  goto error1;
1080  }
1081  continue;
1082  }
1083  } else if ((ct->type == SHARP || ct->type == DIG_SHARP)
1084  && (ls->flags & HANDLE_ASSERTIONS)) {
1085  /* we have an assertion; parse it */
1086  int nnp, ltww = 1;
1087  size_t i = tf1.art;
1088  struct token_fifo atl;
1089  char *aname;
1090  struct assert *a;
1091  int av = 0;
1092  struct token rt;
1093 
1094  atl.art = atl.nt = 0;
1095  while (i < tf1.nt && ttMWS(tf1.t[i].type)) i ++;
1096  if (i >= tf1.nt) goto assert_error;
1097  if (tf1.t[i].type != NAME) goto assert_error;
1098  aname = tf1.t[i ++].name;
1099  while (i < tf1.nt && ttMWS(tf1.t[i].type)) i ++;
1100  if (i >= tf1.nt) goto assert_generic;
1101  if (tf1.t[i].type != LPAR) goto assert_generic;
1102  i ++;
1103  for (nnp = 1; nnp && i < tf1.nt; i ++) {
1104  if (ltww && ttMWS(tf1.t[i].type)) continue;
1105  if (tf1.t[i].type == LPAR) nnp ++;
1106  else if (tf1.t[i].type == RPAR
1107  && (-- nnp) == 0) {
1108  tf1.art = i + 1;
1109  break;
1110  }
1111  ltww = ttMWS(tf1.t[i].type);
1112  aol(atl.t, atl.nt, tf1.t[i], TOKEN_LIST_MEMG);
1113  }
1114  if (nnp) goto assert_error;
1115  if (ltww && atl.nt && (-- atl.nt) == 0) freemem(atl.t);
1116  if (atl.nt == 0) goto assert_error;
1117 
1118  /* the assertion is in aname and atl; check it */
1119  a = get_assertion(aname);
1120  if (a) for (i = 0; i < a->nbval; i ++)
1121  if (!cmp_token_list(&atl, a->val + i)) {
1122  av = 1;
1123  break;
1124  }
1125  rt.type = NUMBER;
1126  rt.name = av ? "1" : "0";
1127  aol(tf2.t, tf2.nt, rt, TOKEN_LIST_MEMG);
1128  if (atl.nt) freemem(atl.t);
1129  continue;
1130 
1131  assert_generic:
1132  tf1.art = i;
1133  rt.type = NUMBER;
1134  rt.name = get_assertion(aname) ? "1" : "0";
1135  aol(tf2.t, tf2.nt, rt, TOKEN_LIST_MEMG);
1136  continue;
1137 
1138  assert_error:
1139  error(l, "syntax error for assertion in #if");
1140  ls->output_fifo = save_tf;
1141  goto error1;
1142  }
1143  aol(tf2.t, tf2.nt, *ct, TOKEN_LIST_MEMG);
1144  }
1145  ls->output_fifo = save_tf;
1146  freemem(tf1.t);
1147  if (tf2.nt == 0) {
1148  error(l, "void condition (after expansion) for a #if/#elif");
1149  return -1;
1150  }
1151 
1152  /*
1153  * suppress whitespace and replace rogue identifiers by 0
1154  */
1155  tf3.art = tf3.nt = 0;
1156  while (tf2.art < tf2.nt) {
1157  struct token *ct = tf2.t + (tf2.art ++);
1158 
1159  if (ttMWS(ct->type)) continue;
1160  if (ct->type == NAME) {
1161  /*
1162  * a rogue identifier; we replace it with "0".
1163  */
1164  struct token rt;
1165 
1166  rt.type = NUMBER;
1167  rt.name = "0";
1168  aol(tf3.t, tf3.nt, rt, TOKEN_LIST_MEMG);
1169  continue;
1170  }
1171  aol(tf3.t, tf3.nt, *ct, TOKEN_LIST_MEMG);
1172  }
1173  freemem(tf2.t);
1174 
1175  if (tf3.nt == 0) {
1176  error(l, "void condition (after expansion) for a #if/#elif");
1177  return -1;
1178  }
1179  eval_line = l;
1180  z = eval_expr(&tf3, &ret, (ls->flags & WARN_STANDARD) != 0);
1181  freemem(tf3.t);
1182  if (ret) return -1;

1184 
1185 error1:
1186  if (tf1.nt) freemem(tf1.t);
1187  if (tf2.nt) freemem(tf2.t);
1188  return -1;
1189 }

1212 #define left_angle(t) ((t) == LT || (t) == LEQ || (t) == LSH \

1236  discard_char(ls);

1299  if (!ttMWS(ls->ctok->type)) {
1300  struct token t;
1301 
1302  t.type = ls->ctok->type;
1303  t.line = l;
1304  if (S_TOKEN(ls->ctok->type)) {
1305  t.name = sdup(ls->ctok->name);

1316  struct token *ct;
